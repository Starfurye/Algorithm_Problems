# BM54 三数之和

## Solution

```javascript
/**
 *
 * @param num int整型一维数组
 * @return int整型二维数组
 */
function threeSum(num) {
    // write code here
    // if (num.length < 3) return [];
    num.sort((a, b) => a - b);
    let ans = [];
    for (let i = 0; i < num.length; i++) {
        if (num[i] > 0) break;
        if (i > 0 && num[i] === num[i - 1]) continue; // 去重

        let left = i + 1,
            right = num.length - 1;
        while (left < right) {
            let sum = num[i] + num[left] + num[right];
            if (sum === 0) {
                ans.push([num[i], num[left++], num[right--]]);
                while (left < right && num[left] === num[left - 1]) left++; // 去重
                while (left < right && num[right] === num[right + 1]) right--; // 去重
            } else if (sum < 0) left++;
            else right--;
        }
    }
    return ans;
}
```
