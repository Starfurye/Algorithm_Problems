# BM4 合并两个排序的链表

## Solution

```js
function Merge(pHead1, pHead2) {
    // write code here
    let head = { next: null };

    let cur = head;
    while (pHead1 && pHead2) {
        if (pHead1.val <= pHead2.val) {
            cur.next = pHead1;
            [cur, pHead1] = [cur.next, pHead1.next];
        } else {
            cur.next = pHead2;
            [cur, pHead2] = [cur.next, pHead2.next];
        }
    }

    if (pHead1) cur.next = pHead1;
    else cur.next = pHead2;
    return head.next;
}
module.exports = {
    Merge: Merge,
};
```
