# BM6 判断链表中是否有环

## Solution

### 快慢指针

```javascript
/*
 * function ListNode(x){
 *   this.val = x;
 *   this.next = null;
 * }
 */

/**
 *
 * @param head ListNode类
 * @return bool布尔型
 */
function hasCycle(head) {
    // write code here
    let slow = head,
        fast = head;
    while (fast) {
        slow = slow.next;
        fast = fast.next;
        if (fast) fast = fast.next;
        else return false;
        if (slow === fast) return true;
    }
    return false;
}
module.exports = {
    hasCycle: hasCycle,
};
```
