# BM9 删除链表的倒数第 n 个节点

## Solution

```javascript
/*
 * function ListNode(x){
 *   this.val = x;
 *   this.next = null;
 * }
 */

/**
 *
 * @param head ListNode类
 * @param n int整型
 * @return ListNode类
 */
function removeNthFromEnd(head, n) {
    // write code here
    if (n === 0 || !head) return head;
    let dh = { next: head };
    let fast = dh,
        slow = dh; // slow在要删的结点前
    for (let i = 0; i < n; i++) {
        if (fast) fast = fast.next;
        else return null;
    }
    while (fast.next) {
        slow = slow.next;
        fast = fast.next;
    }
    slow.next = slow.next.next;
    return dh.next;
}
module.exports = {
    removeNthFromEnd: removeNthFromEnd,
};
```
