# BM7 链表中环的入口结点

## Solution

### 快慢指针

```javascript
function EntryNodeOfLoop(pHead) {
    // write code here
    if (!pHead) return null;
    let slow = pHead,
        fast = pHead;
    while (fast) {
        [slow, fast] = [slow.next, fast.next];
        if (!fast) return null;
        fast = fast.next;
        if (slow === fast) {
            slow = pHead;
            while (slow !== fast) {
                [slow, fast] = [slow.next, fast.next];
            }
            return slow;
        }
    }
    return null;
}
module.exports = {
    EntryNodeOfLoop: EntryNodeOfLoop,
};
```
