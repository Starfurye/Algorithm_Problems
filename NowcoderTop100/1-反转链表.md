# BM1 反转链表

```javascript
function ReverseList(pHead) {
    // write code here
    let pre = null,
        cur = pHead;
    while (cur) {
        let temp = cur.next;
        cur.next = pre;
        [cur, pre] = [temp, cur];
    }
    return pre;
}
module.exports = {
    ReverseList: ReverseList,
};
```
