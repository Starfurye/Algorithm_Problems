# BM92 最长无重复子数组

## Solution

注意在 map 里清除 pre 前的元素。

```javascript
/**
 *
 * @param arr int整型一维数组 the array
 * @return int整型
 */
function maxLength(arr) {
    // write code here
    if (arr.length === 0) return 0;
    let map = new Map();
    let pre = 0,
        after = 1;
    let ans = 1;
    map.set(arr[0], 0);
    while (after < arr.length) {
        if (map.has(arr[after])) {
            let lastPos = map.get(arr[after]);
            // 更新pre前清除map中元素
            for (let i = pre; i <= lastPos; i++) map.delete(arr[i]);
            pre = lastPos + 1;
        }
        map.set(arr[after], after);
        ans = Math.max(ans, after - pre + 1);
        after++;
    }
    return ans;
}
```
