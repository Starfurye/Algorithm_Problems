# BM68 矩阵的最小路径和

## Solution

### 回溯（TLE）

```javascript
/**
 *
 * @param matrix int整型二维数组 the matrix
 * @return int整型
 */
function minPathSum(matrix) {
    // write code here
    const backTrack = (row, col, sum) => {
        if (row === rLen || col === cLen) return; // 到达边界
        if (row === rLen - 1 && col === cLen - 1) {
            ans = Math.min(ans, sum + matrix[row][col]);
            return;
        }
        backTrack(row, col + 1, sum + matrix[row][col]);
        backTrack(row + 1, col, sum + matrix[row][col]);
    };

    let [rLen, cLen] = [matrix.length, matrix[0].length];
    let ans = Number.MAX_SAFE_INTEGER;
    backTrack(0, 0, 0);
    return ans;
}
```

### 动态规划

```javascript
/**
 *
 * @param matrix int整型二维数组 the matrix
 * @return int整型
 */
function minPathSum(matrix) {
    // write code here
    let [rLen, cLen] = [matrix.length, matrix[0].length];
    let dp = new Array(rLen).fill(0).map(() => new Array(cLen).fill(0));

    dp[0][0] = matrix[0][0];
    for (let i = 1; i < cLen; i++) dp[0][i] = dp[0][i - 1] + matrix[0][i];
    for (let i = 1; i < rLen; i++) dp[i][0] = dp[i - 1][0] + matrix[i][0];

    for (let i = 1; i < rLen; i++) {
        for (let j = 1; j < cLen; j++) {
            dp[i][j] = Math.min(dp[i - 1][j], dp[i][j - 1]) + matrix[i][j];
        }
    }
    return dp[rLen - 1][cLen - 1];
}
```

#### 滚动数组优化

```javascript
/**
 *
 * @param matrix int整型二维数组 the matrix
 * @return int整型
 */
function minPathSum(matrix) {
    // write code here
    let [rLen, cLen] = [matrix.length, matrix[0].length];
    let dp = new Array(cLen).fill(0);

    dp[0] = matrix[0][0];
    for (let i = 1; i < cLen; i++) dp[i] = dp[i - 1] + matrix[0][i];

    for (let i = 1; i < rLen; i++) {
        dp[0] += matrix[i][0];
        for (let j = 1; j < cLen; j++) {
            dp[j] = Math.min(dp[j - 1], dp[j]) + matrix[i][j];
        }
    }
    return dp[cLen - 1];
}
```
