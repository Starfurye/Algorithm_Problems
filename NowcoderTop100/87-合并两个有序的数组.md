# BM87 合并两个有序的数组

## Solution

```javascript
/**
 *
 * @param A int整型一维数组
 * @param B int整型一维数组
 * @return void
 */
function merge(A, m, B, n) {
    // write code here
    A.length = m + n;
    let pa = m - 1,
        pb = n - 1,
        p = m + n - 1;
    while (pa >= 0 && pb >= 0) {
        if (A[pa] >= B[pb]) {
            A[p] = A[pa];
            pa--;
        } else {
            A[p] = B[pb];
            pb--;
        }
        p--;
    }
    if (pa < 0) {
        while (p >= 0) {
            A[p] = B[pb];
            p--;
            pb--;
        }
    }
    // pb < 0 时不用处理
}
module.exports = {
    merge: merge,
};
```

## Test example

```javascript
let arrA = [4, 8, 13];
let arrB = [7, 9, 17];
merge(arrA, arrA.length, arrB, arrB.length);
console.log(arrA);
```
