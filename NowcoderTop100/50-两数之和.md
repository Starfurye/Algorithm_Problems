# BM50 两数之和

## Solution

牛客本题的下标返回要求从 1 开始

```javascript
/**
 *
 * @param numbers int整型一维数组
 * @param target int整型
 * @return int整型一维数组
 */
function twoSum(numbers, target) {
    // write code here
    let map = new Map();
    for (let i = 0; i < numbers.length; i++) {
        if (!map.has(target - numbers[i])) {
            map.set(numbers[i], i + 1);
        } else return [map.get(target - numbers[i]), i + 1];
    }
}
module.exports = {
    twoSum: twoSum,
};
```
