# 数组去重

```javascript
let arr = [
    1,
    1,
    "true",
    "true",
    true,
    true,
    15,
    15,
    false,
    false,
    undefined,
    undefined,
    null,
    null,
    NaN,
    NaN,
    "NaN",
    0,
    0,
    "a",
    "a",
    {},
    {},
];
// => [ 1, 'true', true, 15, false, undefined, null, NaN, 'NaN', 0, 'a', {} ]
```

## 终极方案 typeof item + item

```javascript
const unique = (arr) => {
    let map = new Map();
    return arr.filter((item) => {
        if (map.has(typeof item + item)) {
            return false;
        } else {
            map.set(typeof item + item, true);
            return true;
        }
    });
};
```
