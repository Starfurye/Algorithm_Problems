# 柯里化

将接受多个参数的函数变为接受一个参数返回一个函数的形式，便于再次调用。

```javascript
function curry(fn) {
    return function curry1(...args1) {
        if (args1.length >= fn.length) {
            return fn.call(null, ...args1);
        } else {
            return function curry2(...args2) {
                return curry1.apply(null, [...args1, ...args2]);
            };
        }
    };
}
let add = (a, b, c) => a + b + c;
console.log(add(1, 23, 31));
console.log(curry(add)(1)(23, 31));
console.log(curry(add)(1)(23)(31));
// 55
```
