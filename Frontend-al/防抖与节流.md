## 防抖与节流

## 防抖

函数只会执行最后一次，用例：搜索栏、resize 事件。

```html
<!DOCTYPE html>
<html lang="en">
    <head>
        <meta charset="UTF-8" />
        <meta http-equiv="X-UA-Compatible" content="IE=edge" />
        <meta name="viewport" content="width=device-width, initial-scale=1.0" />
        <title>防抖</title>
    </head>
    <body>
        <button>点击</button>
        <script>
            function debounce(fn, time) {
                let timer;
                return function () {
                    clearTimeout(timer);
                    let args = arguments;
                    timer = setTimeout(() => {
                        fn.apply(this, args);
                    }, time);
                };
            }

            function addOne() {
                console.log("clicked!");
            }
            const btn = document.querySelector("button");
            btn.addEventListener("click", debounce(addOne, 3000));
        </script>
    </body>
</html>
```

## 节流

函数每隔一段时间才会执行一次，用例：鼠标点击、滚动事件。

```html
<!DOCTYPE html>
<html lang="en">
    <head>
        <meta charset="UTF-8" />
        <meta http-equiv="X-UA-Compatible" content="IE=edge" />
        <meta name="viewport" content="width=device-width, initial-scale=1.0" />
        <title>节流</title>
        <style>
            .occ {
                background-color: pink;
                height: 800px;
                width: 200px;
            }
        </style>
    </head>
    <body>
        <div class="occ"></div>
        <script>
            function throttle(fn, time) {
                let lastTime = 0;
                return function () {
                    let newTime = new Date();
                    if (newTime - lastTime > time) {
                        fn.apply(this, arguments);
                        lastTime = newTime;
                    }
                };
            }

            function scrollTest() {
                console.log("scollllllll!");
            }
            document.addEventListener("scroll", throttle(scrollTest, 3000));
        </script>
    </body>
</html>
```
