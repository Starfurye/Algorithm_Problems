# 598. Range Addition II

## Solution

### 交集

找出矩阵中所有满足要求的次数**恰好等于**操作次数（即左上角元素的值）的位置。

```javascript
/**
 * @param {number} m
 * @param {number} n
 * @param {number[][]} ops
 * @return {number}
 */
var maxCount = function (m, n, ops) {
    let minrow = m,
        mincol = n;
    for (const op of ops) {
        minrow = Math.min(minrow, op[0]); // 所有操作中最小的行数
        mincol = Math.min(mincol, op[1]); // 所有操作中最小的列数
    }
    return minrow * mincol;
};
```

### 模拟

```javascript
/**
 * @param {number} m
 * @param {number} n
 * @param {number[][]} ops
 * @return {number}
 */
var maxCount = function (m, n, ops) {
    if (ops.length == 0) return m * n;
    // 左上角一定是最大值
    const column = new Array(n).fill(0);
    const row = new Array(m).fill(0);
    for (let op of ops) {
        for (let i = 0; i < op[0]; i++) {
            row[i]++;
        }
        for (let i = 0; i < op[1]; i++) {
            column[i]++;
        }
    }

    let colnum = 1,
        rownum = 1,
        maxnum = Math.min(column[0], row[0]);
    for (let i = 1; i < column.length; i++) {
        if (column[i] == maxnum) colnum++;
    }
    for (let i = 1; i < row.length; i++) {
        if (row[i] == maxnum) rownum++;
    }

    return rownum * colnum;
};
```
